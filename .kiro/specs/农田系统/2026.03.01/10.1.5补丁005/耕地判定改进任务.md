public static bool HasFarmingObstacle(Vector3 cellCenter, string targetSortingLayer)
{
    // 从 FarmTileManager 读取配置
    if (FarmTileManager.Instance == null)
        return false;

    float radius = FarmTileManager.Instance.FarmingObstacleCheckRadius;
    string[] obstacleTags = FarmTileManager.Instance.FarmingObstacleTags;
    string[] whitelistTags = FarmTileManager.Instance.FarmingWhitelistTags;

    // 1. 碰撞体检测（扩大到配置的半径）
    Vector2 boxSize = new Vector2(radius, radius);
    Collider2D[] hits = Physics2D.OverlapBoxAll(cellCenter, boxSize, 0f);

    foreach (var hit in hits)
    {
        // 楼层隔离
        if (!IsSameSortingLayer(hit.transform, targetSortingLayer))
            continue;

        // 白名单检查
        if (whitelistTags.Length > 0 && HasAnyTagStatic(hit.transform, whitelistTags))
            continue;

        // 障碍物标签检查
        if (HasAnyTagStatic(hit.transform, obstacleTags))
            return true;
    }

    // 2. 检测无碰撞体的树苗（1+8 范围）
    if (HasTreeInArea(cellCenter, radius, targetSortingLayer))
        return true;

    // 3. 检测无碰撞体的箱子（1+8 范围）
    if (HasChestInArea(cellCenter, radius, targetSortingLayer))
        return true;

    return false;
}
```

/// <summary>
/// 检查 Transform 是否在目标楼层（通过 SortingLayer 判断）
/// </summary>
private static bool IsSameSortingLayer(Transform t, string targetSortingLayer)
{
    var renderer = t.GetComponentInChildren<SpriteRenderer>();
    if (renderer != null)
    {
        return renderer.sortingLayerName == targetSortingLayer;
    }

    // 如果没有 SpriteRenderer，默认认为在同一层（兼容旧代码）
    return true;
}
```

### B.4 重写 HasTreeInArea 方法

**位置**: 替换原有的 `HasTreeAtPositionStatic` 方法（第232-251行）

```csharp
/// <summary>
/// 检测区域内是否有树木（高效版：一次性获取列表）
/// </summary>
private static bool HasTreeInArea(Vector3 center, float radius, string targetSortingLayer)
{
    // 一次性获取全场景树木列表（避免循环内调用）
    var allTrees = Object.FindObjectsByType<TreeController>(FindObjectsSortMode.None);

    // 使用 AABB 正方形判断（与 Physics2D.OverlapBoxAll 保持一致）
    float halfSize = radius / 2f;

    foreach (var tree in allTrees)
    {
        // 楼层隔离
        if (!IsSameSortingLayer(tree.transform, targetSortingLayer))
            continue;

        // 获取树木位置（使用 Collider bounds 而非 transform.position）
        Vector3 treePos = GetTreePosition(tree);

        // AABB 正方形距离判断（与物理碰撞体检测形状一致）
        if (Mathf.Abs(treePos.x - center.x) <= halfSize && Mathf.Abs(treePos.y - center.y) <= halfSize)
            return true;
    }

    return false;
}

/// <summary>
/// 获取树木位置（优先使用 Collider bounds）
/// </summary>
private static Vector3 GetTreePosition(TreeController tree)
{
    var collider = tree.GetComponentInChildren<Collider2D>();
    if (collider != null)
        return collider.bounds.center;

    // Fallback: 使用父级 Transform（树木根部）
    return tree.transform.parent != null
        ? tree.transform.parent.position
        : tree.transform.position;
}
```

/// <summary>
/// 检测区域内是否有箱子（高效版：一次性获取列表）
/// </summary>
private static bool HasChestInArea(Vector3 center, float radius, string targetSortingLayer)
{
    // 一次性获取全场景箱子列表
    var allChests = Object.FindObjectsByType<ChestController>(FindObjectsSortMode.None);

    // 使用 AABB 正方形判断（与 Physics2D.OverlapBoxAll 保持一致）
    float halfSize = radius / 2f;

    foreach (var chest in allChests)
    {
        // 楼层隔离
        if (!IsSameSortingLayer(chest.transform, targetSortingLayer))
            continue;

        // 获取箱子位置（使用 Collider bounds）
        Vector3 chestPos = GetChestPosition(chest);

        // AABB 正方形距离判断（与物理碰撞体检测形状一致）
        if (Mathf.Abs(chestPos.x - center.x) <= halfSize && Mathf.Abs(chestPos.y - center.y) <= halfSize)
            return true;
    }

    return false;
}

/// <summary>
/// 获取箱子位置（优先使用 Collider bounds）
/// </summary>
private static Vector3 GetChestPosition(ChestController chest)
{
    var collider = chest.GetComponentInChildren<Collider2D>();
    if (collider != null)
        return collider.bounds.center;

    // Fallback: 使用 Transform
    return chest.transform.position;
}

if (requireGroundAround)
{
    for (int dx = -1; dx <= 1; dx++)
    {
        for (int dy = -1; dy <= 1; dy++)
        {
            if (dx == 0 && dy == 0) continue; // 跳过中心格子

            Vector3Int neighborPos = cellPosition + new Vector3Int(dx, dy, 0);
            TileBase neighborTile = tilemaps.groundTilemap.GetTile(neighborPos);

            // 如果周围格子没有地面，不允许耕地
            if (neighborTile == null)
                return false;
        }
    }
}

return true;
